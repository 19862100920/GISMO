0016          end
0017       end

0019

0021       %dataRequest should be a filename
0022       w = loadSacByFilenames(dataRequest);

0026 function w = loadSacByFilenames(filename)
0027    %LOADSAC  creates a waveform from a SAC file
0028    % waveform = loadSacByFilenames(waveform, filename);
0029    % To call, the first argument must be a waveform.  However this waveform
0030    % will never be accessed-- it will be overwritten by the SAC information.
0031    % Therefore, you can use "waveform" as your first argument, so that a
0032    % generic waveform object will be created for you
0033    %
0034    % If the filename is a cell of several SAC file names, such as:
0035    % fn = {'BOB.SHZ.AV','TOM.EHZ.AV','HARRY.EHZ.AV'}
0036    % then, w = loadSacByFilenames(waveform,fn) will return a matrix of waveforms, the
0037    % same size and shape  as fn.   fn(N,M) will be the SAC file name for the
0038    % waveform denoted by w(N,M).  In the example above, loadSacByFilenames will return a
0039    % 1x3 waveform object, with all three files.
0040    %
0041    % Programming suggestion???select files using a dialog box, by using the
0042    % following code:
0043    %   % display the dialog box used for selecting files, and
0044    %   % allow for multiple selections
0045    %   [f,d] = uigetfile({'*.*','all Files (*.*)'}, 'MultiSelect', 'on');
0046    %
0047    %   % f will have one or more file names, so concatenate them
0048    %   % with the directory
0049    %   filename = strcat(d,f);
0050    %
0051    %   %call loadSacByFilenames to load all these files into waveform objects
0052    %   w = loadSacByFilenames(waveform, filename);
0053    %
0054    % The SAC header is parsed out, with the following equivelent fields used
0055    % by waveform:
0056    % _waveform_     _sac_
0057    % STATION        KSTNM
0058    % CHANNEL        KCMPNM
0059    % FREQUENCY        1 / DELTA (ODELTA is not used)
0060    % START            NZYEAR, NZJDAY, NZHOUR, NZMIN, NZSEC, NZMSEC
0061    % UNITS            IDEP is parsed.  IUNK becomes "Counts"
0062    % All header fields with values - that is, header fields that do not
0063    % contain the value  -12345 (either numeric or text) - are put into
0064    % user-defined fields within the waveform.
0065    % These fields can then be accessed via get/set.  User-defined fields
0066    % includes those that are parsed into waveform's predefined  fields as
0067    % listed above.
0068    %
0069    % loadSacByFilenames uses routines originating in code borrowed from
0070    % Mike Thorn
0071    %
0072    % For more information about the SAC header, and the interpretation of the
0073    % fields, check out Lawrence Livermore's web site.  Search for "seismic
0074    % sac" and you'll find it.
0075    
0076    % VERSION: 1.1 of waveform objects
0077    % AUTHOR: Celso Reyes
0078    % $Date$
0079    % $Revision$
0080    
0081    if ~iscell(filename)
0082       filename = {filename};
0083    end
0084    
0085    sacdata = readsacfile(filename{:});
0086    w = sac2waveform(sacdata);
0087    w = removeSacDuplicates(w);
0088    for n=1:numel(filename)
0089       w(n) = addhistory(w(n),'Loaded SAC file: %s',filename{n});
0090    end
0091 end
