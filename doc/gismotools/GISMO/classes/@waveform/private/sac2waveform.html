sac2waveform

PURPOSE

SAC2WAVEFORM fill a waveform object from a sac header

SYNOPSIS

function allWaveforms = sac2waveform(sacStructure)

DESCRIPTION 

SAC2WAVEFORM fill a waveform object from a sac header
   waveform = sac2waveform(sacInfo)
     sacInfo is the output from readsacfile.m

    Each field from the header be either incorporated into the basic
    structure of each waveform or will be added as user-defined fields.

    See also:  waveform/private/readsacfile,
    waveform/private/waveform2sacheader

0003    %   waveform = sac2waveform(sacInfo)
0004    %     sacInfo is the output from readsacfile.m
0005    %
0006    %    Each field from the header be either incorporated into the basic
0007    %    structure of each waveform or will be added as user-defined fields.
0008    %
0009    %    See also:  waveform/private/readsacfile,
0010    %    waveform/private/waveform2sacheader
0011    
0012    % VERSION: 1.1 of waveform objects
0013    % AUTHOR: Celso Reyes (celso@gi.alaska.edu)
0014    %    modified from Michael Thorne (4/2004)  mthorne@asu.edu
0015    % LASTUPDATE: 9/1/2009, improving readability in effort to be able to
0016    % better maintain this code.
0017    
0018    % first test to see if the sacInfo is indeed a sacfile
0019    %---------------------------------------------------------------------------
0020    
0021    allWaveforms = repmat(waveform,size(sacStructure));
0022    
0023    for eachSacFile = 1:numel(sacStructure)
0024       sacInfo = sacStructure(eachSacFile);
0025       
0026       [fieldNames, fieldValues] =  parseHeader(sacInfo.header);
0027       
0028       % Take these results and create a structure with them
0029       %   fieldsPairedWithValues puts it all in an order that "struct()"
0030       %   understands. like so: {fieldname1,fieldval1,...,fieldNameN,fieldvalN}
0031       invalidFields = strcmp(fieldNames,'DUMMYVARIABLE');
0032       fieldNames = fieldNames(~invalidFields);
0033       fieldValues = fieldValues(~invalidFields);
0034       fieldsPairedWithValues = [fieldNames(:)'; fieldValues(:)'];
0035       headerDetails = struct(fieldsPairedWithValues{:});
0036       
0037       
0038       chanTag = getChanneltag(headerDetails);
0039       w = set(waveform,'channelinfo',chanTag);
0040       
0041       for nrecs = 1:numel(fieldNames)
0042          thisfield = fieldNames{nrecs};
0043          thisvalue = fieldValues{nrecs};
0044          if ~isempty(thisvalue)
0045             w = addfield(w,thisfield,headerDetails.(thisfield));
0046          end
0047       end
0048       
0049       w = set(w,'start',getDate(headerDetails));
0050       w = set(w,'data',sacInfo.amplitudes);
0051       w = set(w,'freq', 1 ./ headerDetails.DELTA); %doesn't take into account ODELTA (observed freq)
0052       w = set(w,'units', idep2units(headerDetails.IDEP));
0053       w = clearhistory(w);
0054       allWaveforms(eachSacFile) = w;
0055    end
0056 end

0099 function [headerFields,value] = parseHeader(h)
0100    emptyNumber = -12345;
0101    emptyString = '-12345';
0102    
0103    headerDefinitions = getHeaderDefinitions();
0104    headerFields = headerDefinitions(:,1);
0105    fieldType = headerDefinitions(:,2);
0106    %startPositions = [headerDefinitions{:,3}];
0107    %endPositions = (startPositions - 1) + [ headerDefinitions{:,4}];
0108    value = cell(size(headerFields));
0109    
0110    for thisField=1:numel(headerFields)
0111       %ensure each field is of the right type, and get rid of any default
0112       %values, replacing with '' or [] as appropriate
0113       
0114       value(thisField) = h(thisField);
0115       
0116       switch (fieldType{thisField})
0117          case 'double'
0118             if value{thisField} == emptyNumber
0119                value(thisField) = {[]};
0120             end
0121             
0122          case 'char'
0123             %get rid of all beginning/ending whitespace
0124             value(thisField) = {strtrim(char(value{thisField}))};
0125             
0126             if strcmpi(value(thisField),emptyString)
0127                value(thisField) = {''};
0128             end
0129             
0130          case 'integer'
0131             value(thisField) = {round(value{thisField})};
0132             if value{thisField} == emptyNumber
0133                value(thisField) = {[]};
0134             end
0135             
0136          case 'logical'
0137             if strcmp(value{thisField},emptyString)
0138                value(thisField) = {false};
0139             end
0140             value(thisField) = {logical(value{thisField})};
0141             
0142          otherwise
0143             error('Waveform:sac2waveform:unknownHeaderTypedef',...
0144                'Unknown type definition within the SAC header');
0145       end
0146    end
0147 end
